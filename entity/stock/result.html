Result

Once you've got the result you should always check if an error has occurred, if not, retrieve the resource.

use Railken\Amethyst\Managers\StockManager;

$manager = new StockManager();

$result = $manager->create([
    "warehouse" => [
        "name" => "Reagan Hammes",
        "description" => "Illo sit et ipsa. Possimus doloribus et voluptatum quis et et nostrum. Repudiandae est accusantium odio voluptatem omnis enim."
    ],
    "key" => 10,
    "stock" => 10,
    "stockable_type" => "foo",
    "stockable" => [
        "name" => "Delia Greenholt",
        "description" => "Occaecati eveniet iste atque impedit non est. Doloribus consequatur est voluptas maiores rerum. Omnis enim ut alias quis non expedita.",
        "bar" => [
            "name" => "Mittie Kozey DDS",
            "description" => "Nesciunt officia laborum cumque qui quis quisquam est. Optio placeat sunt voluptatem aut ut distinctio sequi id. Quidem omnis expedita delectus in blanditiis. Quo pariatur qui mollitia molestiae."
        ]
    ]
]);

if ($result->ok()) {

    $resource = $result->getResource();

} else {

    // Loop through all errors
    $result->getErrors()->map(function($error) {
        return $error->toArray();
    }))

    // Retrieve an array of all errors
    $result->getSimpleErrors();

    /* The result is something like this:

        [0] => Array
            (
                [code] => FIELD_NOT_DEFINED
                [label] => field
                [message] => The field is required
                [value] =>
            )
    */

}
